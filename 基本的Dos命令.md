```bash
dir                 --->查看当前路径下的所有目录
cd                   --->切换目录
cd /d f:            --->切换到f盘
cd ..               --->到上一级
cls                 --->清除屏幕
exit                --->退出终端
ipconfig            --->查看ip
ping www.baidu.com  --->查看百度的IP地址
md 文件夹名         --->创建文件夹
rd 文件夹名         --->移除文件夹
del 文件名          --->删除文件
cd> 文件名          --->生成文件
```
